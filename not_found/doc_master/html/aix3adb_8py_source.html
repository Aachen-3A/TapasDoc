241 class cookietransportrequest:
242  """A Transport request method that retains cookies over its lifetime.
243 
244  The regular xmlrpclib transports ignore cookies. Which causes
245  a bit of a problem when you need a cookie-based login, as with
246  the Bugzilla XMLRPC interface.
247 
248  So this is a helper for defining a Transport which looks for
249  cookies being set in responses and saves them to add to all future
250  requests.
251  """
252  # From http://www.lunch.org.uk/wiki/xmlrpccookies
253  # Inspiration drawn from
254  # http://blog.godson.in/2010/09/how-to-make-python-xmlrpclib-client.html
255  # http://www.itkovian.net/base/transport-class-for-pythons-xml-rpc-lib/
256  #
257  # Note this must be an old-style class so that __init__ handling works
258  # correctly with the old-style Transport class. If you make this class
259  # a new-style class, Transport.__init__() won't be called.
260 
261  cookies = []
262  def setcookies(self,cookiefile, domain):
263  self.cookies = []
264  jar=cookielib.MozillaCookieJar(cookiefile)
265  jar.load(ignore_discard=False, ignore_expires=False)
266  for cookie in jar:
267  if cookie.domain==domain:
268  self.cookies.append("$Version=1; "+cookie.name+"="+cookie.value+";")
269  def send_cookies(self, connection):
270  if self.cookies:
271  for cookie in self.cookies:
272  connection.putheader("Cookie", cookie)

288  # Deal with differences between Python 2.4-2.6 and 2.7.
289  # In the former h is a HTTP(S). In the latter it's a
290  # HTTP(S)Connection. Luckily, the 2.4-2.6 implementation of
291  # HTTP(S) has an underlying HTTP(S)Connection, so extract
292  # that and use it.
293  try:

295  except AttributeError:
